19static void serialize_lua_object( sol::object val, JsonOut &jsout, std::vector<sol::table> &stack )
20{
21 sol::state_view lua( val.lua_state() );
22
23 jsout.start_object();
24 switch( val.get_type() ) {
25 case sol::type::boolean: {
26 jsout.member_as_string( "type", "bool" );
27 jsout.member( "data" );
28 jsout.write( val.as<bool>() );
29 break;
30 }
31 case sol::type::number: {
32 // There's no clear difference in JSON between int and float,
33 // so we have to be explicit to avoid subtle errors down the line
34 if( is_number_integer( lua, val ) ) {
35 jsout.member_as_string( "type", "int" );
36 jsout.member( "data" );
37 jsout.write( val.as<int64_t>() );
38 } else {
39 jsout.member_as_string( "type", "float" );
40 jsout.member( "data" );
41 jsout.write( val.as<double>() );
42 }
43 break;
44 }
45 case sol::type::string: {
46 jsout.member_as_string( "type", "string" );
47 jsout.member( "data" );
48 jsout.write( val.as<std::string>() );
49 break;
50 }
51 case sol::type::table: {
52 jsout.member_as_string( "type", "lua_table" );
53 jsout.member( "data" );
54 serialize_lua_table_internal( val.as<sol::table>(), jsout, stack );
55 break;
56 }
57 case sol::type::userdata: {
58 std::optional<std::string> luna_type = get_luna_type( val );
59 if( !luna_type ) {
60 debugmsg( "Tried to serialize usertype that was not registered with luna." );
61 break;
62 }
63 sol::table table_val = val.as<sol::table>();
64 sol::protected_function serialize_func = table_val["serialize"];
65 if( !serialize_func ) {
66 debugmsg( "Tried to serialize usertype that does not allow serialization." );
67 break;
68 }
69 jsout.member_as_string( "type", "userdata" );
70 jsout.member_as_string( "kind", *luna_type );
71 jsout.member( "data" );
72 sol::protected_function_result res = serialize_func( val, jsout );
73 if( res.status() != sol::call_status::ok ) {
74 sol::error err = res;
75 debugmsg( "Failed to serialize type '%s': %s", *luna_type, err.what() );
76 }
77 break;
78 }
79 default: {
80 // TODO: throw error
81 debugmsg( "Unsupported type encountered when serializing Lua table." );
82 break;
83 }
84 }
85 jsout.end_object();
86}

88void serialize_lua_table_internal( sol::table t, JsonOut &jsout, std::vector<sol::table> &stack )
89{
90 for( const sol::table &it : stack ) {
91 if( it == t ) {
92 debugmsg( "Tried to serialize recursive table structure." );
93 jsout.write_null();
94 return;
95 }
96 }
97
98 stack.push_back( t );
99
100 sol::state_view lua( t.lua_state() );
101 jsout.start_object();
102
103 if( !t.empty() ) {
104 jsout.member( "entries" );
105 jsout.start_array();
106
107 // TODO: persistent key order?
108 t.for_each( [&]( sol::object key, sol::object val ) {
109 serialize_lua_object( key, jsout, stack );
110 serialize_lua_object( val, jsout, stack );
111 } );
112
113 jsout.end_array();
114 }
115 jsout.end_object();
116
117 stack.pop_back();
118}

126static sol::object
127deserialize_lua_object( sol::state_view lua, const JsonObject &jo )
128{
129 std::string entry_type = jo.get_member( "type" );
130 if( entry_type == "string" ) {
131 std::string data = jo.get_string( "data" );
132 return sol::object( lua, sol::in_place, data );
133 } else if( entry_type == "bool" ) {
134 bool data = jo.get_bool( "data" );
135 return sol::object( lua, sol::in_place, data );
136 } else if( entry_type == "int" ) {
137 int data = jo.get_int( "data" );
138 return sol::object( lua, sol::in_place, data );
139 } else if( entry_type == "float" ) {
140 double data = jo.get_float( "data" );
141 return sol::object( lua, sol::in_place, data );
142 } else if( entry_type == "lua_table" ) {
143 JsonObject data = jo.get_object( "data" );
144 sol::table new_table = lua.create_table();
145 deserialize_lua_table( new_table, data );
146 return new_table;
147 } else if( entry_type == "userdata" ) {
148 std::string kind = jo.get_member( "kind" );
149 JsonIn &data_raw = *jo.get_raw( "data" );
150 // Horrible hack ahead
151 std::string script = string_format( "return %s.new()", kind );
152 sol::protected_function_result res = lua.script( script, "deserialize_init", sol::load_mode::any );
153 if( res.status() != sol::call_status::ok ) {
154 debugmsg( "Failed to init container for deserializable type '%s'", kind );
155 } else {
156 sol::object obj = res;
157 sol::table obj_table = obj.as<sol::table>();
158 if( !obj_table ) {
159 debugmsg( "Failed to init container for deserializable type '%s'", kind );
160 } else {
161 sol::protected_function deserialize_func = obj_table["deserialize"];
162 if( !deserialize_func ) {
163 debugmsg( "Failed to deserialize type '%s': not deserializable.", kind );
164 } else {
165 sol::protected_function_result res = deserialize_func( obj, data_raw );
166 if( res.status() != sol::call_status::ok ) {
167 sol::error err = res;
168 debugmsg( "Failed to deserialize type '%s': %s", kind, err.what() );
169 } else {
170 return obj;
171 }
172 }
173 }
174 }
175 } else {
176 debugmsg( "Deserialization of record type '%s' is not implemented.", entry_type );
177 }
178 return sol::nil;
179}

181void deserialize_lua_table( sol::table t, JsonObject &obj )
182{
183 sol::state_view lua( t.lua_state() );
184
185 if( !obj.has_array( "entries" ) ) {
186 return;
187 }
188
189 JsonArray arr = obj.get_array( "entries" );
190 if( arr.size() % 2 != 0 ) {
191 debugmsg( "invalid array size %d", arr.size() );
192 return;
193 }
194 for( size_t idx = 0; idx < arr.size(); idx += 2 ) {
195 JsonObject jo_key = arr.get_object( idx );
196 JsonObject jo_val = arr.get_object( idx + 1 );
197 sol::object key = deserialize_lua_object( lua, jo_key );
198 sol::object val = deserialize_lua_object( lua, jo_val );
199 t.set( key, val );
200 }
201}
